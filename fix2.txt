   echo -e “duck\nmoose\ncat” > README.md

   git checkout feature1
 
5. sed -i ‘’ ’s/moose/mouse/‘ README.md

   git add . 
   git commit -m “Change moose to mouse and add dog in the last line”

6. git checkout main  
   sed -i ‘’ ’s/moose/mammoth/' README.md 
   git add . 
   git commit -m “Change moose to mammoth”

7. git merge feature1 
   git status 
   git diff 
   open -e README.md (fixed the conflict manually in the file) 
   git add README.md   
   git commit -m “Conflict solved and merge all animals”

8. git tag extinct
